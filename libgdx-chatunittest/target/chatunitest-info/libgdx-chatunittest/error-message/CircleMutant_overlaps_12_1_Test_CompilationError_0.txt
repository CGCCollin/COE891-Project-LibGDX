class CircleMutant_overlaps_12_1_Test {

    private final Class<CircleMutant> circleClass = CircleMutant.class;

    @BeforeEach
    public void setUp() throws Exception {
        // Create an instance of the focal class
        circleInstance = circleClass.newInstance();
    }

    @Test
    public void testOverlapsPositiveCases() throws Exception {
        // Test overlapping circles with different positions and radii
        CircleMutant c1 = new CircleMutant(0, 0, 5);
        CircleMutant c2 = new CircleMutant(3, 4, 6);
        assertTrue(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertTrue(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
        // Test overlapping circles with same position and different radii
        c1 = new CircleMutant(0, 0, 5);
        c2 = new CircleMutant(0, 0, 6);
        assertTrue(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
        // Test overlapping circles with same position and radii
        c1 = new CircleMutant(0, 0, 5);
        c2 = new CircleMutant(0, 0, 5);
        assertTrue(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
    }

    @Test
    public void testOverlapsNegativeCases() throws Exception {
        // Test non-overlapping circles with different positions and radii
        CircleMutant c1 = new CircleMutant(0, 0, 5);
        CircleMutant c2 = new CircleMutant(6, 8, 10);
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
        // Test non-overlapping circles with same position and different radii
        c1 = new CircleMutant(0, 0, 5);
        c2 = new CircleMutant(0, 0, 10);
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
        // Test non-overlapping circles with same position and radii
        c1 = new CircleMutant(0, 0, 5);
        c2 = new CircleMutant(0, 0, 5);
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c1));
        assertFalse(circleClass.getMethod("overlaps", CircleMutant.class).invoke(circleInstance, c2));
    }
}
